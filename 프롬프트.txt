def _detect_sector_type(sector: str, industry: str = "") -> str:
    """
    sector/industry 문자열 → 내부 섹터 타입 식별.

    yfinance sector 값 예시: "Technology", "Financial Services",
    "Consumer Cyclical", "Healthcare", "Energy", "Industrials" 등
    DART industry 값: 한국 업종코드 문자열
    """
    text = f"{sector} {industry}".lower()

    if any(k in text for k in [
        "semiconductor", "반도체", "memory", "메모리", "wafer", "chip",
        "electronic component", "전자부품",
    ]):
        return "semiconductor"

    if any(k in text for k in [
        "bank", "financial service", "insurance", "capital market",
        "금융", "은행", "보험", "증권", "신용", "저축",
    ]):
        return "financial"

    if any(k in text for k in [
        "auto", "vehicle", "자동차", "automotive", "motor company",
    ]):
        return "automotive"

    if any(k in text for k in [
        "pharma", "biotech", "drug", "의약", "바이오", "제약",
    ]):
        return "pharma"

    if any(k in text for k in [
        "oil", "gas", "energy", "power", "utility", "에너지",
        "발전", "정유", "석유", "가스",
    ]):
        return "energy"

    if any(k in text for k in [
        "battery", "배터리", "chemical", "화학", "material", "소재",
        "lithium", "리튬",
    ]):
        return "battery_materials"

    if any(k in text for k in [
        "consumer", "retail", "food", "beverage", "소비", "유통",
        "식품", "백화점", "의류",
    ]):
        return "consumer"

    if any(k in text for k in [
        "software", "internet", "saas", "it service", "소프트웨어",
        "플랫폼", "data processing", "cloud service",
    ]):
        return "software"

    if any(k in text for k in [
        "telecom", "communication service", "통신", "wireless", "mobile network",
    ]):
        return "telecom"

    if any(k in text for k in [
        "defense", "aerospace", "방산", "항공", "무기", "방위",
    ]):
        return "defense"

    return "default"


def _build_system_prompt(sector_type: str) -> str:
    """섹터 타입 → 완성된 시스템 프롬프트 (섹터 전문 지식 + 공통 규칙)."""
    sector_context = SECTOR_SYSTEM_PROMPTS.get(sector_type, SECTOR_SYSTEM_PROMPTS["default"])
    return sector_context + _BASE_RULES
